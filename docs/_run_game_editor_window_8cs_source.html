260  private async Task RunBatch()
261  {
262  Debug.Log($"Running batch with {this.NumberOfBatchRunClients} clients," +
263  $" {this.NumberOfBatchRunServers} servers and with editor mode: {this.BatchRunEditorMode}");
264 
265  List<GameRunner> runners = new();
266 
267  if (this.NumberOfBatchRunClients > 0 || this.NumberOfBatchRunServers > 0)
268  {
269  GameBuilder builder = new();
270  builder.IsDebug = true;
271 
272  var (path, success) = builder.Build(this.BuildPath, "BatchBuild");
273  if (!success)
274  {
275  Debug.LogError("Cannot run batch because batch build failed.");
276  }
277 
278  Debug.Log($"Batch build built to path: `{path}`");
279 
280  for (var i = 0; i < this.NumberOfBatchRunClients; ++i)
281  {
282  GameRunner runner = new(path);
283  runner.NetworkRunState = NetworkRunStates.Client;
284  runner.NetworkConnectTarget = NetworkConnectTargets.Localhost;
285  runner.Run();
286 
287  runners.Add(runner);

289 
290  for (var i = 0; i < this.NumberOfBatchRunServers; ++i)
291  {
292  GameRunner runner = new(path);

294  runner.Run();
295

297  }
298  }

300  if (this.BatchRunEditorMode == BatchRunEditorModes.Client)
301  {
302  this.RunKitchenSceneAsClient();
303  }
304  else if (this.BatchRunEditorMode == BatchRunEditorModes.Server)
